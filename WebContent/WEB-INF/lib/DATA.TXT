CREATE TABLE COMPLAINT ( COMPLAINTNO INT UNSIGNED NOT NULL AUTO_INCREMENT,PRIMARY KEY (COMPLAINTNO),USERNAME VARCHAR(30) NOT NULL,REQTYPE VARCHAR(30),CATEGORY VARCHAR(30),EMAIL VARCHAR(40),ISSUE VARCHAR(MAX));

CREATE TABLE DETAIL(SEQNO INT UNSIGNED NOT NULL AUTO_INCREMENT, TYPE VARCHAR(20), AMOUNT INT(40), DATE VARCHAR(40), USER1 VARCHAR(30),USER2 VARCHAR(30),BALANCE(40));
